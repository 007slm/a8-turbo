-- ========= 参数：本次要新增的行数 =========
SET @batch := 100000;  -- 例如 10 行；想要 100万就写 1000000

-- ========= 会话与批处理设置（测试环境可用） =========
SET @old_fk_checks := @@FOREIGN_KEY_CHECKS; SET FOREIGN_KEY_CHECKS = 0;
SET @old_unique    := @@UNIQUE_CHECKS;      SET UNIQUE_CHECKS    = 0;
SET @old_ai        := @@AUTOCOMMIT;         SET AUTOCOMMIT       = 0;

-- ========= 计算续跑区间 =========
-- 从当前最大 sSequence 后续跑
SET @start := (SELECT COALESCE(MAX(sSequence), 0) + 1 FROM sa_task);
SET @end   := @start + @batch - 1;

-- 本批次的全局种子（保证本批次内以及与历史批次都不冲突）
SET @seed := REPLACE(UUID(), '-', '');

INSERT INTO sa_task (

  sFlowID, sSourceID, sFrontID,
  sExecuteMode2, sExecuteMode, sPreemptMode,

  sImportanceID, sImportanceName, sEmergencyID, sEmergencyName,

SELECT
  -- ✅ 主键：批次种子 + 行号，避免与历史/并发批次冲突
  LOWER(MD5(CONCAT(@seed, '-', n)))                                     AS sID,

  CONCAT('TASK-', LPAD(n, 8, '0'))                                      AS sCode,
  NULL                                                                  AS sParentID,    -- 本批次末尾再 UPDATE 构造层级
  CONCAT('Task #', n)                                                   AS sName,
  CONCAT('Auto generated content for task ', n)                         AS sContent,
  NULL                                                                  AS sRemark,
  NULL                                                                  AS sSummary,

  SUBSTRING(REPLACE(UUID(),'-',''), 1, 32)                              AS sFlowID,
  NULL                                                                  AS sSourceID,
  NULL                                                                  AS sFrontID,

  ELT((n % 3)+1, 'serial', 'parallel', 'pool')                          AS sExecuteMode2,
  ELT((n % 2)+1, 'auto', 'manual')                                      AS sExecuteMode,
  ELT((n % 2)+1, 'normal', 'preempt')                                   AS sPreemptMode,

  CONCAT('cat-',  n % 200)                                              AS sCatalogID,
  ELT((n % 3)+1, 'kindA', 'kindB', 'kindC')                             AS sKindID,
  CONCAT('type-', n % 50)                                               AS sTypeID,
  CONCAT('Type ', n % 50)                                               AS sTypeName,

  CONCAT('imp-',  n % 5)                                                AS sImportanceID,
  ELT((n % 5)+1, '低', '较低', '中', '较高', '高')                       AS sImportanceName,
  CONCAT('emg-',  n % 5)                                                AS sEmergencyID,
  ELT((n % 5)+1, '不紧急', '较不紧急', '一般', '较紧急', '紧急')           AS sEmergencyName,

  CONCAT('process_',  n % 100)                                          AS sProcess,
  CONCAT('activity_', n % 50)                                           AS sActivity,
  CONCAT('/c/', n)                                                      AS sCURL,
  CONCAT('/e/', n)                                                      AS sEURL,

  -- 时间分布：起点随 n 递增
  TIMESTAMP('2024-01-01 00:00:00') + INTERVAL n MINUTE                  AS sCreateTime,
  TIMESTAMP('2024-01-01 00:00:00') + INTERVAL n MINUTE + INTERVAL 5 MINUTE                 AS sDistributeTime,
  TIMESTAMP('2024-01-01 00:00:00') + INTERVAL n MINUTE + INTERVAL (n % 120) MINUTE         AS sLastModifyTime,
  TIMESTAMP('2024-01-01 00:00:00') + INTERVAL n MINUTE + INTERVAL (n % 240) MINUTE         AS sWarningTime,

  TIMESTAMP('2024-01-01 00:00:00') + INTERVAL n MINUTE + INTERVAL (n % 480) MINUTE         AS sLimitTime,
  (n % 86400)                                                           AS sLimitSecond,
  TIMESTAMP('2024-01-01 00:00:00') + INTERVAL n MINUTE + INTERVAL 60 MINUTE                AS sExpectStartTime,
  TIMESTAMP('2024-01-01 00:00:00') + INTERVAL n MINUTE + INTERVAL 180 MINUTE               AS sExpectFinishTime,
  TIMESTAMP('2024-01-01 00:00:00') + INTERVAL n MINUTE + INTERVAL (n % 30) MINUTE          AS sActualStartTime,
  TIMESTAMP('2024-01-01 00:00:00') + INTERVAL n MINUTE + INTERVAL (30 + n % 240) MINUTE    AS sActualFinishTime,
  TIMESTAMP('2024-01-01 00:00:00') + INTERVAL n MINUTE + INTERVAL (n % 60) MINUTE          AS sExecuteTime,

  CONCAT('U', LPAD(n % 10000, 4, '0'))                                  AS sCreatorPersonID,
  CONCAT('创建者', n % 10000)                                            AS sCreatorPersonName,
  CONCAT('pos',  n % 200)                                               AS sCreatorPosID,
  CONCAT('岗位', n % 200)                                               AS sCreatorPosName,
  CONCAT('dept', n % 100)                                               AS sCreatorDeptID,
  CONCAT('部门', n % 100)                                               AS sCreatorDeptName,
  CONCAT('org',  n % 20)                                                AS sCreatorOgnID,
  CONCAT('机构', n % 20)                                                AS sCreatorOgnName,
  CONCAT('/org', n%20, '/dept', n%100, '/pos', n%200, '/u', LPAD(n%10000,4,'0'))          AS sCreatorFID,
  CONCAT('/机构',n%20,'/部门',n%100,'/岗位',n%200,'/用户',LPAD(n%10000,4,'0'))             AS sCreatorFName,

  CONCAT('U', LPAD((n+1) % 10000, 4, '0'))                               AS sExecutorPersonID,
  CONCAT('执行者', (n+1) % 10000)                                        AS sExecutorPersonName,
  CONCAT('pos',  (n+3) % 200)                                            AS sExecutorPosID,
  CONCAT('岗位', (n+3) % 200)                                            AS sExecutorPosName,
  CONCAT('dept', (n+7) % 100)                                            AS sExecutorDeptID,
  CONCAT('部门', (n+7) % 100)                                            AS sExecutorDeptName,
  CONCAT('org',  (n+9) % 20)                                             AS sExecutorOgnID,
  CONCAT('机构', (n+9) % 20)                                             AS sExecutorOgnName,
  CONCAT('/org', (n+9)%20, '/dept', (n+7)%100, '/pos', (n+3)%200, '/u', LPAD((n+1)%10000,4,'0')) AS sExecutorFID,
  CONCAT('/机构',(n+9)%20,'/部门',(n+7)%100,'/岗位',(n+3)%200,'/用户',LPAD((n+1)%10000,4,'0'))  AS sExecutorFName,
  CONCAT('执行者', (n+1)%100, ',执行者', (n+2)%100)                        AS sExecutorNames,
  CONCAT('活动', n % 50, ',活动', (n+1) % 50)                             AS sActivityNames,

  ELT((n % 2)+1, 'Y', 'N')                                              AS sResponsible,
  CONCAT('CUST', n % 50000)                                             AS sCustomerID,
  CONCAT('C', LPAD(n % 50000, 5, '0'))                                  AS ScustomerCode,
  CONCAT('客户', n % 50000)                                             AS sCustomerName,

  CONCAT('PRJ',  LPAD(n % 2000, 4, '0'))                                AS sProjectID,
  CONCAT('PRJ-', LPAD(n % 2000, 4, '0'))                                AS sProjectCode,
  CONCAT('项目', n % 2000)                                              AS sProjectName,

  CONCAT('PLAN',  n % 5000)                                             AS sPlanID,
  CONCAT('PL-',   n % 5000)                                             AS sPlanCode,
  CONCAT('计划',  n % 5000)                                             AS sPlanName,

  LPAD(n % 1000, 3, '0')                                                AS sData1,
  LPAD((n/3) % 1000, 3, '0')                                            AS sData2,
  LPAD((n/7) % 1000, 3, '0')                                            AS sData3,
  LPAD((n/11)% 1000, 3, '0')                                            AS sData4,

  ELT((n % 2)+1, 'N', 'Y')                                              AS sFake,
  ELT((n % 2)+1, 'Y', 'N')                                              AS sActive,

  ELT((n % 4)+1, 'TODO','RUNNING','DONE','CANCELLED')                   AS sStatusID,
  ELT((n % 4)+1, '待办','执行中','已完成','已取消')                       AS sStatusName,

  NULL                                                                  AS sAIID,
  NULL                                                                  AS sAIStatusID,
  NULL                                                                  AS sAIStatusName,
  ELT((n % 2)+1, 'Y', 'N')                                              AS sAIActive,

  (n % 10000)                                                           AS sWorkTime,
  CONCAT('v', n % 100)                                                  AS sVersionNumber,
  n                                                                     AS sSequence,   -- ✅ 从 @start 续跑
  NULL                                                                  AS sLock,

  NULL                                                                  AS sHints,
  NULL                                                                  AS sShortcut,
  ELT((n % 3)+1, 'none','email','sms')                                  AS sRemindMode,

  (n % 10)                                                              AS version,
  (n % 2)                                                               AS sWithdraw,

  NULL                                                                  AS sVariable,
  CONCAT('Process ',  n % 100)                                          AS sProcessName,
  CONCAT('Activity ', n % 50)                                           AS sActivityName,

  NULL                                                                  AS sProcessTemplateID,
  NULL                                                                  AS sProcessTemplateItemSequence,
  NULL                                                                  AS sTempPermissionID,
  NULL                                                                  AS sActivityInTemplate,
  NULL                                                                  AS sProcessTemplateID2,

  CONCAT('ES01-', n % 1000), CONCAT('ES02-', n % 1000),
  CONCAT('ES03-', n % 1000), CONCAT('ES04-', n % 1000),
  CONCAT('ES05-', n % 1000), CONCAT('ES06-', n % 1000),
  CONCAT('ES07-', n % 1000), CONCAT('ES08-', n % 1000),

  (n % 1000), (n % 1000), (n % 1000), (n % 1000),

  TIMESTAMP('2024-01-01') + INTERVAL (n % 100000) SECOND,
  TIMESTAMP('2024-01-01') + INTERVAL (n % 200000) SECOND,
  TIMESTAMP('2024-01-01') + INTERVAL (n % 300000) SECOND,
  TIMESTAMP('2024-01-01') + INTERVAL (n % 400000) SECOND,

  CONCAT('ET31-', n % 1000), CONCAT('ET32-', n % 1000),
  CONCAT('ET33-', n % 1000), CONCAT('ET34-', n % 1000),

  (n % 10000), ((n+1) % 10000), ((n+2) % 10000), ((n+3) % 10000),

  NULL, NULL, NULL, NULL  -- LONGBLOB 不造大对象（需要时再按比例 UPDATE）
FROM (
  -- 生成 0..999999 的连续数字 n（若你要 >100万行，再多加一层 0..9 CROSS JOIN）
  SELECT
    (a.n + 10*b.n + 100*c.n + 1000*d.n + 10000*e.n + 100000*f.n) AS n
  FROM
    (SELECT 0 n UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4
                 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) a
  CROSS JOIN
    (SELECT 0 n UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4
                 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) b
  CROSS JOIN
    (SELECT 0 n UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4
                 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) c
  CROSS JOIN
    (SELECT 0 n UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4
                 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) d
  CROSS JOIN
    (SELECT 0 n UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4
                 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) e
  CROSS JOIN
    (SELECT 0 n UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4
                 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) f
) seq
WHERE n BETWEEN @start AND @end
ORDER BY n;

-- ========= 构造父子层级（每 10 条一个组头）=========
-- 以本批次起点 @start 为基准对齐分组，避免续跑时错位
UPDATE sa_task t
JOIN sa_task p
  ON p.sSequence = (FLOOR((t.sSequence - @start)/10) * 10) + @start
 AND p.sSequence BETWEEN @start AND @end
SET t.sParentID = p.sID
WHERE t.sSequence BETWEEN @start AND @end
  -- 组头自己的 sParentID 也会被设置为自身组头，这里允许（如需避免可再加条件：AND t.sSequence <> p.sSequence）
  AND t.sParentID IS NULL;

-- ========= 提交并恢复会话设置 =========
COMMIT;
SET FOREIGN_KEY_CHECKS = @old_fk_checks;
SET UNIQUE_CHECKS    = @old_unique;
SET AUTOCOMMIT       = @old_ai;

FROM sa_task t
         JOIN (
    SELECT 'morning' AS slot_name
    UNION ALL SELECT 'afternoon'
    UNION ALL SELECT 'evening'
) slot
         JOIN (
    SELECT 0 i UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4
    UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9
) d ON 1=1  -- 乘以 10
